  <script>
    // ===== Configuración (sustituye por tus claves/ajustes) =====
    const CONFIG = {
      shopPassword: 'gnostic-shop',         // <— REEMPLAZA
      empalmePassword: 'empalme-vid',       // <— REEMPLAZA (segunda clave)
      rotatorPhrases: [
        'Love is Magic',
        'Knowledge is Magic',
        'All is Magic',
        'Everything is Magic'
      ],
      rotatorMs: 2600
    };

    // Año footer
    document.getElementById('year').textContent = new Date().getFullYear();

    // Rotador accesible (respeta reduced motion)
    (function phraseRotator(){
      const el = document.getElementById('rotator');
      const phrases = CONFIG.rotatorPhrases;
      if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) { el.textContent = phrases[0]; return; }
      let i = 0; setInterval(()=>{ i = (i+1)%phrases.length; el.style.opacity = 0; setTimeout(()=>{ el.textContent = phrases[i]; el.style.opacity = 1; }, 180); }, CONFIG.rotatorMs);
    })();

    // Gate Shop (demo en cliente)
    function openShop(e){ e.preventDefault();
      const pass = document.getElementById('shop-pass').value.trim();
      const ok = pass === CONFIG.shopPassword;
      document.getElementById('shop-error').style.display = ok ? 'none' : 'block';
      if(ok){ document.getElementById('shop-gate').style.display='none'; document.getElementById('catalog').style.display='block'; }
      return false;
    }

    // Mindprint modal
    function openMindprint(){ document.getElementById('modal-mindprint').showModal(); }

    // Empalme gate + vídeo
    function openEmpalmeGate(){ document.getElementById('modal-empalme-gate').showModal(); }
    function openEmpalme(e){ e.preventDefault();
      const pass = document.getElementById('empalme-pass').value.trim();
      const ok = pass === CONFIG.empalmePassword;
      document.getElementById('empalme-error').style.display = ok ? 'none' : 'block';
      if(ok){ closeModal('modal-empalme-gate'); document.getElementById('modal-empalme-video').showModal(); }
      return false;
    }

    // Utilidades
    function closeModal(id){ const dlg = document.getElementById(id); if(dlg && dlg.open) dlg.close(); }
    function fakeSubmit(e){ e.preventDefault(); alert('Mensaje enviado (demo). Sustituye por tu backend.'); return false; }

    // Marcar enlace actual de navegación según el hash
    function markCurrent(){
      document.querySelectorAll('.menu a').forEach(a=>a.removeAttribute('aria-current'));
      const id = location.hash || '#home';
      const link = document.querySelector(`.menu a[href="${id}"]`);
      if(link) link.setAttribute('aria-current','page');
    }
    window.addEventListener('hashchange', markCurrent); markCurrent();
  </script>
